class Tee:
    def __init__(self, *streams):
        self.streams = streams
    def write(self, data):
        for s in self.streams:
            try:
                s.write(data)
            except Exception:
                pass
    def flush(self):
        for s in self.streams:
            try:
                s.flush()
            except Exception:
                pass

# Helper to save plots automatically
PLOT_INDEX = 0
def _slug(s: str) -> str:
    s = s.strip()
    s = re.sub(r"\s+", "_", s)
    s = re.sub(r"[^\w\-\.]+", "", s)
    return s or "plot"

def savefig_show(extra_name: str = ""):
    """Save current figure into OUTPUT_DIR with incremental index, using its title if available, then show."""
    global PLOT_INDEX
    fig = plt.gcf()
    title = ""
    if fig.axes:
        try:
            title = fig.axes[0].get_title() or ""
        except Exception:
            title = ""
    PLOT_INDEX += 1
    base = f"{PLOT_INDEX:03d}_{_slug(title)}"
    if extra_name:
        base += "_" + _slug(extra_name)
    out_path = os.path.join(OUTPUT_DIR, base + ".png")
    plt.savefig(out_path, dpi=200, bbox_inches="tight")
    print(f"[PLOT SAVED] {out_path}")
    plt.show()

def save_table(df: pd.DataFrame, filename: str):
    """Save a dataframe as CSV into the OUTPUT_DIR and log it."""
    path = os.path.join(OUTPUT_DIR, filename)
    df.to_csv(path, index=False)
    print(f"[TABLE SAVED] {path}")

# HELPERS
def fmt(x, d=DECIMALS):
    try:
        return f"{x:.{d}f}"
    except Exception:
        return str(x)

def print_header(title):
    bar = "-" * max(20, len(title))
    print(f"\n{bar}\n{title}\n{bar}")

def print_year_summary(year, wcc_nodes, scc_nodes, n_communities, nmi=None):
    line = f"Year {year}: WCC={wcc_nodes} nodes | SCC={scc_nodes} nodes | Communities={n_communities}"

# Community detection (Leiden)
def leiden_partition_directed_weighted(G_dir, weight_attr="weight", seed=42, resolution=None):
    try:
        import igraph as ig
        import leidenalg as la
    except ImportError as e:
        raise ImportError("Please install igraph and leidenalg: pip install igraph leidenalg") from e

    nodes = list(G_dir.nodes())
    idx = {n: i for i, n in enumerate(nodes)}

    g = ig.Graph(directed=True)
    g.add_vertices(len(nodes))
    g.vs["name"] = nodes

    edges = [(idx[u], idx[v]) for u, v in G_dir.edges()]
    g.add_edges(edges)

    weights = [float(G_dir[u][v].get(weight_attr, 1.0)) for u, v in G_dir.edges()]
    g.es["weight"] = weights

    if resolution is None:
        part = la.find_partition(g, la.ModularityVertexPartition, weights=g.es["weight"], seed=seed)
    else:
        part = la.find_partition(g, la.CPMVertexPartition, weights=g.es["weight"],
                                 resolution_parameter=resolution, seed=seed)

    membership = part.membership
    return {node: comm for node, comm in zip(nodes, membership)}

# ========= Builders (aggregation χωρίς iterrows) =========
def aggregate_edges_df(df_sub: pd.DataFrame) -> pd.DataFrame:
    """
    Επιστρέφει aggregated dataframe ανά (SOURCE,TARGET) με:
      weight=sum(|rating|), rating_sum, edge_count, rating=mean, cost=1/(weight+1),
      first_time, last_time
    """
    grp = (df_sub
           .assign(abs_rating=lambda x: x['RATING'].abs())
           .groupby(['SOURCE', 'TARGET'], as_index=False)
           .agg(
               weight=('abs_rating', 'sum'),
               rating_sum=('RATING', 'sum'),
               edge_count=('RATING', 'size'),
               first_time=('TIME', 'min'),
               last_time=('TIME', 'max'),
           ))
    grp['rating'] = grp['rating_sum'] / grp['edge_count']
    grp['cost'] = 1.0 / (grp['weight'] + 1.0)
    return grp

def build_digraph_from_aggregated_df(grp: pd.DataFrame) -> nx.DiGraph:
    """Χτίζει DiGraph από aggregated dataframe (μία ακμή ανά (u,v))."""
    Gx = nx.DiGraph()
    Gx.add_edges_from([
        (int(row['SOURCE']), int(row['TARGET']),
         {
            'weight': float(row['weight']),
            'rating_sum': float(row['rating_sum']),
            'count': int(row['edge_count']),   # use edge_count to avoid Series.count clash
            'rating': float(row['rating']),    # mean rating
            'cost': float(row['cost']),
            'first_time': row['first_time'],
            'last_time': row['last_time']
         })
        for _, row in grp.iterrows()
    ])
    return Gx

#  Faster betweenness for large graphs
def betweenness_cost_weighted(Gdir: nx.DiGraph):
    n = Gdir.number_of_nodes()
    if n > 1500:
        k = min(300, n)  # approximation for speed
        return nx.betweenness_centrality(Gdir, weight="cost", normalized=True, k=k, seed=42)
    return nx.betweenness_centrality(Gdir, weight="cost", normalized=True)

#  Robustness helpers
def avg_spl_scc_cost_exact(Gdir: nx.DiGraph):
    """Μέσο μήκος συντομότερου μονοπατιού στο μεγαλύτερο SCC (cost-weighted)."""
    if Gdir.number_of_nodes() == 0:
        return float("nan")
    try:
        scc_nodes = max(nx.strongly_connected_components(Gdir), key=len)
    except ValueError:
        return float("nan")
    H = Gdir.subgraph(scc_nodes).copy()
    if H.number_of_nodes() < 2:
        return float("nan")
    return nx.average_shortest_path_length(H, weight="cost")

def score_nodes(Gdir, strategy, rnd=None):
    if Gdir.number_of_nodes() == 0:
        return {}
    if strategy == "betweenness":
        return nx.betweenness_centrality(Gdir, weight="cost", normalized=True)
    elif strategy == "in_strength":
        return dict(Gdir.in_degree(weight="weight"))
    elif strategy == "out_strength":
        return dict(Gdir.out_degree(weight="weight"))
    elif strategy == "random":
        if rnd is None:
            rnd = random.Random(42)
        return {n: rnd.random() for n in Gdir.nodes()}
    else:
        raise ValueError("Unknown strategy")

def normalized_trapz(y):
    m = len(y)
    if m == 0:
        return float('nan')
    if m == 1:
        return float(y[0])
    s = 0.0
    for i in range(1, m):
        s += 0.5 * (y[i-1] + y[i])
    return s / (m - 1)

def run_targeted_attack(G_original, strategy,
                        adaptive=ADAPTIVE, max_steps=MAX_STEPS,
                        stop_frac=STOP_FRAC, rng_seed=42, tag=None):
    G = G_original.copy()
    rnd = random.Random(rng_seed)
    try:
        base_wcc = len(max(nx.weakly_connected_components(G), key=len))
    except ValueError:
        base_wcc = 0
    try:
        base_scc = len(max(nx.strongly_connected_components(G), key=len))
    except ValueError:
        base_scc = 0

    print_header(f"Attack strategy: {strategy}{(' — ' + tag) if tag else ''}")
    print(f"Baseline WCC={base_wcc} nodes | SCC={base_scc} nodes")

    scores = None if adaptive else score_nodes(G, strategy, rnd)
    rows = []
    for step in range(1, min(max_steps, G.number_of_nodes()) + 1):
        if G.number_of_nodes() == 0:
            print("No nodes left to remove.")
            break
        if adaptive:
            scores = score_nodes(G, strategy, rnd)
        if not scores:
            print("No nodes to score/remove.")
            break

        node, node_score = max(scores.items(), key=lambda kv: (kv[1], str(kv[0])))
        if node not in G:
            scores.pop(node, None)
            continue

        G.remove_node(node)

        try:
            wcc_size = len(max(nx.weakly_connected_components(G), key=len))
        except ValueError:
            wcc_size = 0
        try:
            scc_size = len(max(nx.strongly_connected_components(G), key=len))
        except ValueError:
            scc_size = 0

        wcc_frac = (wcc_size / base_wcc) if base_wcc else 0.0
        scc_frac = (scc_size / base_scc) if base_scc else 0.0
        asp_cost = avg_spl_scc_cost_exact(G)

        rows.append({
            "step": step,
            "removed_node": node,
            "removed_score": node_score,
            "nodes_remaining": G.number_of_nodes(),
            "edges_remaining": G.number_of_edges(),
            "wcc_nodes": wcc_size,
            "wcc_fraction": wcc_frac,
            "scc_nodes": scc_size,
            "scc_fraction": scc_frac,
            "asp_cost": asp_cost,
        })

        print(
            f"Step {step:>3} | removed node {node} (score={fmt(node_score)})  "
            f"| WCC {wcc_size} ({fmt(wcc_frac)})  "
            f"| SCC {scc_size} ({fmt(scc_frac)})  "
            f"| ASP(SCC,cost) {fmt(asp_cost)}"
        )

        if (wcc_frac <= stop_frac) or (scc_frac <= stop_frac) or (wcc_size <= 1) or (scc_size <= 1):
            print(f"Stopped: break condition met (≤ {int(stop_frac*100)}% of baseline).")
            break

        if not adaptive:
            scores.pop(node, None)

    df_attack = pd.DataFrame(rows)
    auc_wcc = normalized_trapz(df_attack["wcc_fraction"].tolist())
    auc_scc = normalized_trapz(df_attack["scc_fraction"].tolist())
    return df_attack, auc_wcc, auc_scc

#  Null models (rewire + attribute re-assignment)
def _directed_rewire_preserve_degree(G_in: nx.DiGraph, rewire_per_edge=5, seed=42):
    """
    Προσπαθεί degree-preserving rewiring με directed_edge_swap.
    Επιστρέφει νέο DiGraph με ίδια nodes/edges, διαφορετική τοπολογία.
    """
    H = nx.DiGraph()
    H.add_nodes_from(G_in.nodes())
    H.add_edges_from(G_in.edges())  # ξεκινάμε από την ίδια τοπολογία

    try:
        from networkx.algorithms.swap import directed_edge_swap
        m = H.number_of_edges()
        nswap = max(1, rewire_per_edge * m)
        directed_edge_swap(H, nswap=nswap, max_tries=nswap * 10, seed=seed)  # in-place
        return H
    except Exception:
        # Fallback: στοχαστικό stub-matching
        rng = random.Random(seed)
        out_deg = {n: G_in.out_degree(n) for n in G_in.nodes()}
        in_deg  = {n: G_in.in_degree(n)  for n in G_in.nodes()}
        sources = [u for u, k in out_deg.items() for _ in range(k)]
        targets = [v for v, k in in_deg.items()  for _ in range(k)]
        if len(sources) != len(targets):
            raise ValueError("Stub counts mismatch in fallback rewiring.")

        for _ in range(10):
            rng.shuffle(sources); rng.shuffle(targets)
            edges = set(); ok = True
            for u, v in zip(sources, targets):
                if u == v or (u, v) in edges:
                    ok = False; break
                edges.add((u, v))
            if ok:
                H = nx.DiGraph()
                H.add_nodes_from(G_in.nodes())
                H.add_edges_from(edges)
                return H

        # Αποδοχή λίγων συγκρούσεων – επιδιόρθωση
        pairs = list(zip(sources, targets))
        rng.shuffle(pairs)
        H = nx.DiGraph(); H.add_nodes_from(G_in.nodes())
        for u, v in pairs:
            if u != v and not H.has_edge(u, v):
                H.add_edge(u, v)
        while H.number_of_edges() < G_in.number_of_edges():
            u = rng.choice(list(H.nodes())); v = rng.choice(list(H.nodes()))
            if u != v and not H.has_edge(u, v):
                H.add_edge(u, v)
        return H

def _attributes_multiset(G_in: nx.DiGraph):
    """Μάζεψε τα attribute tuples από τις ακμές για re-assignment."""
    attrs = []
    for _, _, d in G_in.edges(data=True):
        attrs.append({
            'weight': d['weight'],
            'rating': d.get('rating', 0.0),
            'rating_sum': d.get('rating_sum', d.get('rating', 0.0)),
            'count': d.get('count', 1),
            'first_time': d.get('first_time', None),
            'last_time': d.get('last_time', None),
        })
    return attrs

def _assign_attributes_from_bag(G_topo: nx.DiGraph, attr_bag: list, seed=42):
    """Σε κάθε ακμή του G_topo δίνουμε ένα τυχαίο attribute dict από το attr_bag (ίδια κατανομή)."""
    rng = random.Random(seed)
    bag = attr_bag[:]
    rng.shuffle(bag)
    if len(bag) < G_topo.number_of_edges():
        raise ValueError("Attribute bag smaller than edges.")
    i = 0
    for u, v in G_topo.edges():
        a = bag[i]; i += 1
        cost = 1.0 / (float(a['weight']) + 1.0)
        G_topo[u][v].update({
            'weight': float(a['weight']),
            'rating': float(a['rating']),
            'rating_sum': float(a['rating_sum']),
            'count': int(a['count']),
            'first_time': a['first_time'],
            'last_time': a['last_time'],
            'cost': float(cost),
        })
    return G_topo

def make_null_graphs_for_year(G_year: nx.DiGraph, n_nulls=8, rewire_per_edge=5, seed=123):
    """
    Επιστρέφει λίστα από n_nulls DiGraphs «null» για ένα έτος.
    - Ίδιοι κόμβοι / ίδιο πλήθος ακμών (και, όπου είναι δυνατό, ίδιο degree-sequence)
    - Διαφορετική τοπολογία (rewire)
    - Ίδια κατανομή weight/rating (ανατεθειμένη τυχαία στις νέες ακμές)
    """
    bag = _attributes_multiset(G_year)
    nulls = []
    for i in range(n_nulls):
        H_topo = _directed_rewire_preserve_degree(G_year, rewire_per_edge, seed + i)
        H_attr = _assign_attributes_from_bag(H_topo, bag, seed + 10_000 + i)
        nulls.append(H_attr)
    return nulls
